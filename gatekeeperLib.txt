--// [CORE FUNCTIONS] \\--
local function calculateUNCScore(passed, total, percentage)
    UNC_Rating.Total = total
    UNC_Rating.Passed = passed
    UNC_Rating.Percentage = percentage
    
    if #UNC_Rating.TotalFailed > 0 then
        return false, UNC_Rating
    end

    return true, UNC_Rating
end

--// [UNC TESTING FUCNTIONS] \\--
Function_Library.Functions.UNC.UNC_Test = function()
    local testsPassed = 0

    for _, functionName in ipairs(UNC_Rating.FunctionsToTest) do
        if getfenv()[functionName] or _G[functionName] or shared[functionName] then
            testsPassed = testsPassed + 1
        else
            table.insert(UNC_Rating.TotalFailed, functionName)
        end
    end

    return calculateUNCScore(testsPassed, #UNC_Rating.FunctionsToTest, (testsPassed / #Static_UNC_Results.FunctionsToTest) * 100)
end

--// [AUTHORIZATION FUNCTIONS] \\--
Function_Library.Functions.Authorization.Check_License = function()
    local success, response = pcall(function()
        return request({
            Url = Server_URL,
            Method = 'POST',
            Headers = { ['Content-Type'] = 'application/json' },
            Body = HttpService:JSONEncode({
                API_KEY = User_Config.License_ID,
                HARDWARE_ID = User_Config.Hardware_ID,
                USER_ID = User_Config.User_ID
            })
        })
    end)
    
    if not success then return false, 'Offline' end

    if response.StatusCode == 200 then
        return true, 'Authorized'
    elseif response.StatusCode == 401 then
        return false, 'Unauthorized'
    else
        return false, 'Error', response.StatusCode
    end
end
